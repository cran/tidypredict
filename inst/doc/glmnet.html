parsnip

parsnip fitted models are also supported by tidypredict:

library(parsnip)

p_model <- linear_reg(penalty = 1) %>%
  set_engine("glmnet") %>%
  fit(mpg ~ ., data = mtcars)
tidypredict_fit(p_model)
#> 35.3140536966127 + (cyl * -0.871623418095165) + (hp * -0.0101157918502673) + 
#>     (wt * -2.59426484734253)
